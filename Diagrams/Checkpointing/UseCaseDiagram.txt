@startuml
actor "Data Source" as DataSource
actor "Data Sync" as DataSync

rectangle "Apache Flink" {
  usecase "Processed Logic" as ProcessedLogic
  usecase "Side Effected Logic" as SideEffectedLogic
  usecase "Processed State" as ProcessedState
}

usecase "Event Driven Application Checkpointing" as Checkpointing

DataSource --> Checkpointing : Send Data
Checkpointing --> ProcessedLogic : Processed Logic
Checkpointing --> SideEffectedLogic : Side Effected Logic
Checkpointing --> ProcessedState : Processed State
ProcessedState --> Checkpointing : Update State
Checkpointing --> DataSync : Send Processed Data
DataSync --> Checkpointing : Receive Processed Data
@enduml
